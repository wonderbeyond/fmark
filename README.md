## Quick setup:

    $ chmod +x ./fmark.py
    $ ln -s `pwd`/fmark.py ~/bin/fmark
    $ echo "source `pwd`/alias.sh" >> ~/.bashrc
    $ fmark -h

## aliases:

    alias fmadd="fmark add"
    alias fmget="fmark get"
    alias fmls="fmark ls"
    alias fmll="fmark ls -l"
    alias fmdel="fmark del"
    alias fmedit="fmark get -e"
